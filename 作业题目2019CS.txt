题3.71不用做，答案如下，供参考学习：
#define BUFSIZE 64
void good_echo()
{
 char buf[BUFSIZE];
 int i;
 while (1) {
	 if (!fgets(buf, BUFSIZE, stdin))
	 return; /* End of file or error */
	 
	 /* Print characters in buffer */
	 for (i = 0; buf[i] && buf[i] != ’\n’; i++)
	 if( putchar(buf[i]) == EOF)
	 return; /* Error */
	 
	 if (buf[i] == ’\n’) {
		 /* Reached terminating newline */
		 putchar(’\n’);
		 return;
	 }
  }
}
